program : 			FUNCTION ID LPAREN args RPAREN function | FUNCTION ID LPAREN args RPAREN function program
decl_args : 		ID | ID COMMA decl_args
function :			FUNCTION LBRACK statement_list RBRACK | LBRACK statement_list RBRACK function
call_args:			expr | expr COMMA call_args
function_call : 	ID LPAREN call_args RPAREN
statement_list : 	statement
               		| statement SEMI statement_list
statement : 		compound_statement
          			| assignment_statement
          			| if_block
          			| empty
if_block: IF LPAREN expr RPAREN LBRACK statement_list RBRACK | IF LPAREN expr RPAREN LBRACK statement_list RBRACK else_block
else_block: ELSE LBRACK statement_list RBRACK | ELSE if_block
assignment_statement : ID ASSIGN expr | ID LBRACK INTEGER RBRACK ASSIGN expr
empty :
expr: term ((PLUS | MINUS) term)*
term: factor ((MUL | DIV) factor)*
factor : PLUS factor
       | MINUS factor
       | INTEGER
       | LPAREN expr RPAREN
       | variable
	   | function
	   | STRING
comparaison: expr (LT | LE | GT | GE | EQ) expr
variable: ID
